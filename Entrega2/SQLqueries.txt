1. Muestre el nombre de todas las navieras.
SELECT nav_nombre FROM Naviera;

2. Muestre todos los buqeus de la naviera 'Francis Drake S.A.'.
SELECT Buque.* FROM Buque, Naviera WHERE Buque.nav_id = Naviera.nav_id;

3. Muestre todos los buques que hayan atracado en 'Valparaiso' el 2020.
SELECT Buque.* FROM Buque, Atraque, Puerto WHERE Puerto.puerto_nombre = 'Valparaiso' AND (YEAR(Atraque.fecha_atraque) = 2020 OR YEAR(Atraque.fecha_salida) = 2020) AND Atraque.puerto_id = Puerto.puerto_id AND Buque.buq_id = Atraque.buq_id;

4. Muestre todos los buques que hayan estado en 'Mejillones' al mismo tiempo que el buque 'Magnolia'.
SELECT Buque.* FROM (SELECT * FROM Buques, Puerto, Atraque WHERE Puerto.puerto_nombre = 'Mejillones' AND Buque.buq_nombre = 'Magnolia' AND Atraque.buq_id = Buque.buq_id) AS Foo, Buque, Atraque WHERE ((Atraque.fecha_atraque >= Foo.fecha_atraque AND Atraque.fecha_atraque <= Foo.fecha_salida) OR (Atraque.fecha_salida >= Foo.fecha_atraque AND Atraque.fecha_salida <= Foo.fecha_salida)) AND Atraque.buq_id <> Foo.buq_id AND Buque.buq_id = Atraque.buq_id;
